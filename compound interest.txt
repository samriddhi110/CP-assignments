#include <stdio.h>
#include <math.h>

int main()
{
    float p, r, t ,c;
    
    printf("enter the principle interest");
    scanf("%f" , &p);
    printf("enter the rate of interest");
    scanf("%f" , &r);
    printf("enter the number of years");
    scanf("%f" , &t);
    
     c = p *(pow(1 + r/100 , t));
    printf(" the compound interest : %f" , c);

    return 0;
}
